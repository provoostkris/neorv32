69uint32_t __attribute__ ((noinline)) riscv_intrinsic_clz(uint32_t rs1) {
70
71 register uint32_t result __asm__ ("a0");
72 register uint32_t tmp_a __asm__ ("a0") = rs1;
73
74 // dummy instruction to prevent GCC "constprop" optimization
75 asm volatile ("add x0, %[input_i], x0" : : [input_i] "r" (tmp_a));
76
77 // clz a0, a0
78 CUSTOM_INSTR_R1_TYPE(0b0110000, 0b00000, a0, 0b001, a0, 0b0010011);
79
80 return result;
81}

510uint32_t riscv_emulate_clz(uint32_t rs1) {
511
512 uint32_t sreg = rs1;
513 uint32_t cnt = 0;
514
515 while(1) {
516 if (sreg & 0x80000000UL) {
517 break;
518 }
519 else {
520 sreg <<= 1;
521 cnt++;
522 }
523 }
524
525 return cnt;
526}

536

538 uint32_t cnt = 0;

540 while(1) {
541 if (sreg & 1) {
542 break;

546 cnt++;
547 }
548 }

550 return cnt;

560uint32_t riscv_emulate_cpop(uint32_t rs1) {
561
562 uint32_t sreg = rs1;
563 uint32_t cnt = 0;
564 int i;
565
566 for (i=0; i<32; i++) {
567 if (sreg & 1) {
568 cnt++;
569 }
570 sreg >>= 1;
571 }
572
573 return cnt;
574}

749uint32_t riscv_emulate_rol(uint32_t rs1, uint32_t rs2) {
750
751 uint32_t shamt = rs2 & 0x1f;
752
753 uint32_t tmp_a = rs1 << shamt;
754 uint32_t tmp_b = rs1 >> (32-shamt);
755
756 return tmp_a | tmp_b;
757}

784uint32_t riscv_emulate_rev8(uint32_t rs1) {
785
786 uint32_t tmp_a = (rs1 & 0x000000ffUL) << 24;
787 uint32_t tmp_b = (rs1 & 0x0000ff00UL) << 8;
788 uint32_t tmp_c = (rs1 & 0x00ff0000UL) >> 8;
789 uint32_t tmp_d = (rs1 & 0xff000000UL) >> 24;
790
791 return tmp_a | tmp_b | tmp_c | tmp_d;
792}

802
803 uint32_t tmp = 0;
804
805 if (rs1 & 0x000000ffUL) {
806 tmp |= 0x000000ffUL;
807 }
808 if (rs1 & 0x0000ff00UL) {
809 tmp |= 0x0000ff00UL;
810 }
811 if (rs1 & 0x00ff0000UL) {
812 tmp |= 0x00ff0000UL;
813 }
814 if (rs1 & 0xff000000UL) {
815 tmp |= 0xff000000UL;
816 }
817
818 return tmp;
819}
